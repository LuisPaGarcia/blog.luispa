I﻿n JS is quite of easy to replace chars by using `str.replace` for example, but, what happens when you want to remove accent marks?

T﻿his tiny post will build an example on how to make a slug from a Spanish sentence.

```javascript

const spanishSentences = [
  "teléfono",
  "árbol",
  "colibrí",
  "cómpramelo",
  "Mandó una carta",
  "Qué grande es ese perro"
];

function createSlug(input){
  return input
    .toLowerCase()
    .trim()
    .replace(/[\s_-]+/g, "-")
    .replace(/^-+|-+$/g, "")
    .normalize("NFD")
    .replace(/[\u0300-\u036f]/g, "");
}

const output = spanishSentences.map(createSlug);
console.log(output);
```
